C语言中的输出函数

C语言代码的框架​

#include <stdio.h>   //头文件

int main()   //主函数
{
    //代码
    return 0;   //结束函数的运行
}

C语言中的输出函数​

printf函数​

printf("字符串 + 格式控制串",参数1，参数2，...);

功能: 向屏幕上输出双引号中的内容，双引号中字符串原样输出，格式控制串会被后面的参数给替换掉，参数的个数由格式控制串的个数来决定.

参数

格式控制串功能
%d输出十进制数，把后面参数当作十进制数据输出
%c输出字符
%s输出字符串
%f输出小数
%e以科学计数法的形式输出小数 ，3.14e10
%o输出八进制数，若是加上#，会输出对应的标志位0
%x输出十六进制数据，若是加上#，会输出对应的标志位0x

示例代码1​

//01-printf.c
#include <stdio.h>

int main()
{
        printf("int = %d %d %d\n",10,20,30);
        printf("char = %c %c %c\n",'X','Y','Z');
        printf("string = %s\n","welcome to china!");
        printf("float = %f\n",3.14159267);
        printf("float = %e\n",131492834.23234323);
        printf("=================================\n");

        printf("dec = %d\n",10); //十进制
        printf("oct = %#o\n",10); //八进制
        printf("hex = %#x\n",10); //十六进制

        return 0;
}

运行结果​

int = 10 20 30
char = X Y Z
string = welcome to china!
float = 3.141593
float = 1.314928e+08
=================================
dec = 10
oct = 012
hex = 0xa

示例代码2​

//02-printf.c
#include <stdio.h>

// '\数据'  数据以八进制形式展示 
// '\x数据' 数据以十六进制形式展示 
int main()
{
        printf("%c = %d\n",'A','A');
        printf("%c = %d\n",'A' + 32,'A' + 32);
        printf("%c = %d\n",'1','1' - 48); //单个字符输出整数的形式
        printf("%c = %d\n",'\141','\141'); //数据以八进制形式输出
        printf("%c = %d\n",'\x61','\x61'); //数据以十六进形式输出

        return 0;
}

运行结果​

A = 65
a = 97
1 = 1
a = 97
a = 97

示例代码3​

//03-define.c
#include <stdio.h>
#define N  10
#define M  N + N
#define SUM  M * M

//宏定义：在数据之中，宏定义是原样替换
int main()
{
        printf("M = %d\n",M);  //N + N--->10 + 10

        printf("SUM = %d\n",SUM);//M * M --->N + N * N + N--->10 + 10 * 10 + 10    120

        return 0;
}

运行结果​

M = 20
SUM = 120

练习​

练习1​

  • 要求利用"%c : %d" 这种格式，向屏幕上输出'a'，'m'，' '对应的字符形式和十进制数形式。
  • 要求利用%f输出3.14159287
  • 要求利用%e输出31455452232.88232
  • printf("data1 = %c : %d",?,?); //?中包含'A'思考如何变成'a'
  • printf("data2 = %d", ?); //?中包含'1'，思考如何通过'1'要求输出十进制数1

代码​

#include <stdio.h>

int main()
{
    printf("%c: %d\n", 'a', 'a');
    printf("%c: %d\n", 'm', 'm');
    printf("%c: %d\n", ' ', ' ');

    printf("%f\n", 3.14159287);

    printf("%e\n", 31455452232.88232);

    printf("data1 = %c : %d\n", 'A' + 32, 'A' + 32);

    printf("data2 = %d\n", '1' - 48);
    return 0;
}

运行结果​

a: 97
m: 109
 : 32
3.141593
3.145545e+10
data1 = a : 97
data2 = 1
