1-7 学生管理系统之更新学生信息

一、功能说明​

  • 在之前框架的基础上实现update_student(head)的功能
  • 该函数主要功能为更新已经存在的学生的信息，具体功能如下：
    • 设计update_student()函数，让用户输入要修改学生的信息。遍历链表找到对应的id,调用update_student_info()函数准备更新对应结点的学生。
      • 设计update_student_info()函数，内部调用display_update_info()函数显示界面。判断用户输入的编号是否合理。然后调用user_change_data()修改用户输入的数据
        • display_update_info()函数，显示1～6编号信息，对应分别为修改学号，修改姓名，修改性别，修改年龄，修改分数，退出
        • user_change_data()函数，则根据用户选择的编号，修改链表结点中的数据

#ifndef EMBEDDED_LINKLIST_H
#define EMBEDDED_LINKLIST_H

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 学生信息结构体
typedef struct student {
    int id; // 学号
    char name[20]; // 姓名
    char sex[20]; // 性别
    int age; // 年纪
    double score; // 分数
} s_t;

typedef s_t datatype_t;

typedef struct node {
    datatype_t data; // 数据域：保存有效数据
    struct node *next; // 指针域：保存下一个结点的地址
} linklist_t;

extern linklist_t *create_empty_linklist();

extern void insert_tail_linklist(linklist_t *head, datatype_t data);

extern int is_empty_linklist(linklist_t *head);

extern int delete_data_linklist(linklist_t *head, int id);

extern void clean_up_linklist(linklist_t **head);

#endif //EMBEDDED_LINKLIST_H

linklist.c​

//
// Created by gaosx on 22-8-31.
//
#include "linklist.h"

// 创建空的链表---为头结点在堆区分配空间
linklist_t *create_empty_linklist() {
    linklist_t *head = NULL;

    // 1.1 分配堆区空间
    head = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == head) {
        printf("malloc failed!\n");
        return NULL;
    }

    memset(head, 0, sizeof(linklist_t));
    // head->next = NULL;
    // head->data = 0;
    return head;
}

// 尾插法
// 特点：插入的顺序和输出的顺序是相同的
void insert_tail_linklist(linklist_t *head, datatype_t data) {
    // 3.1 为结点在堆区申请空间
    linklist_t *temp = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == temp) {
        printf("malloc failed!\n");
    }

    // 3.2 插入数据
    temp->data = data;

    // 3.3 找到尾结点
    linklist_t *tail = head;
    while (tail->next != NULL) {
        tail = tail->next;
    }

    // 3.4 连接结点
    temp->next = tail->next;
    tail->next = temp;
}

int is_empty_linklist(linklist_t *head) {
    return head->next == NULL;
}

int delete_data_linklist(linklist_t *head, int id) {
    linklist_t *p = NULL;
    linklist_t *q = NULL;

    int flag = 0;

    if (is_empty_linklist(head)) {
        return -1;
    }

    p = head;

    // 没有遍历到链表尾部
    while (p->next != NULL) {
        if (p->next->data.id == id) {
            // 保存要删除节点的地址
            q = p->next;
            p->next = q->next;
            free(q);
            q = NULL;

            flag = 1;
        } else {
            p = p->next;
        }
    }

    if (flag == 0) {
        return -2;
    }
    return 0;
}

// 删除链表中所有的结点
void clean_up_linklist(linklist_t **head) {
    linklist_t *p = *head;
    linklist_t *q = NULL;
    while (p != NULL) {
        q = p->next;
        free(p);
        p = NULL;
        p = q;
    }
    *head = NULL;
}

void insert_student(linklist_t *head) {
    s_t st = {0};
    bool is_id_exist = false;
    linklist_t *p = NULL;

    // 1. 请输入学生的信息
    printf("------------------------------------\n");
    printf("请输入学生的信息【学号 姓名 性别 年龄 分数】：");
    scanf("%d%s%s%d%lf", &(st.id), st.name, st.sex, &(st.age), &(st.score));

    while (getchar() != '\n'); // 清除缓冲区

    // 2. 遍历链表，以id为基准，判断学生是否存在
    p = head;
    while (p->next != NULL) {
        if (p->next->data.id == st.id) {
            is_id_exist = true;
            printf("已经存在id=%d号学生，请重新输入！\n", st.id);
            break;
        }
        p = p->next;
    }
    if (!is_id_exist) {
        insert_tail_linklist(head, st);
        printf("插入成功\n");
    }
}

void display_student(linklist_t *head) {
    linklist_t *p = head;

    printf("==========================学生信息表==========================\n");
    printf("------------------------------------------------------------\n");
    printf("|学号\t| 姓名\t| 性别\t| 年龄\t| 分数\t|\n");
    while (p->next != NULL) {
        printf("|%d\t| %s\t| %s\t| %d\t| %.1f\t|\n", p->next->data.id, \
        p->next->data.name, p->next->data.sex, p->next->data.age, p->next->data.score);
        p = p->next;
    }
    printf("------------------------------------------------------------\n");
}

void user_change_data(linklist_t *node, int id) {
    char user_data[100] = {0};
    int new_id;
    int new_age;
    double new_score;

    printf("请输入要修改的数据：");
    scanf("%s", user_data);

    switch (id) {
        case 1:
            new_id = atoi(user_data);
            if (new_id > 0) {
                node->data.id = new_id;
            }
            break;
        case 2:
            memset(node->data.name, 0, sizeof(node->data.name));
            strcpy(node->data.name, user_data);
            break;
        case 3:
            memset(node->data.sex, 0, sizeof(node->data.sex));
            strcpy(node->data.sex, user_data);
            break;
        case 4:
            new_age = atoi(user_data);
            if (new_age > 0) {
                node->data.age = new_age;
            }
            break;
        case 5:
            new_score = atof(user_data);
            if (new_score > 0) {
                node->data.score = new_score;
            }
            break;
        case 6:
            printf("退出\n");
            break;
    }

    printf("修改成功!\n");
}

void update_student_info(linklist_t *node) {
    int num = -1;
    display_update_info();

    loop:
    printf("请输入你要修改的学生信息编号[1~6]：");
    scanf("%d", &num);

    while (getchar() != '\n');

    if (!(num >= 1 && num <= 6)) {
        printf("信息输入错误，请重试！\n");
        goto loop;
    }

    user_change_data(node, num);
}

void update_student(linklist_t *head) {
    linklist_t *p = head;
    int id = 0;

    printf("请输入要修改的学生编号：");
    scanf("%d", &id);

    while (getchar() != '\n');

    while (p->next != NULL) {
        if (p->next->data.id == id) {
            update_student_info(p->next);
            return;
        }
        p = p->next;
    }

    printf("------------------------------------------------------------\n");
    printf("编号输入错误！\n");
    printf("------------------------------------------------------------\n");
}
