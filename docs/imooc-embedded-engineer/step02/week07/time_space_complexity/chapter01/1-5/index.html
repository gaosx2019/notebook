1-5 表达式求值简单版

栈在表达式计算过程中的应用 ：建立操作数栈和运算符栈。运算符有优先级。规则：

自左至右扫描表达式，凡是遇到操作数一律进操作数栈。

当遇到运算符时，如果它的优先级比运算符栈栈顶元素的优先级高就进栈。反之，取出栈顶运算符和操作数栈栈顶的连续两个操作数进行运算，并将结果存入操作数栈，然后继续比较该运算符与栈顶运算符的优先级。

思考：如果我想要用栈实现下列公式的计算，该怎么办？
(注：这里先不考虑空格问题)
//1+123*5;
char buf[1024];

思考："123"如何转换成整型数123？
答案：
int data = 0;
for(i = 0;i < 3;i++){
data = data * 10 + buf[i] - '';
}


思路：
1、创建两个栈
linkstack * operand ;//运算数栈
linkstack * operator; //运算符栈

2、扫描表达式。
<1>若是运算数,合并成一个完整的数data。
规则：直接入操作数栈
push_stack(operator_stack,data);

<2>若是操作符（考虑优先级问题）
规则：1、栈为空，直接入栈
                  2、当前的运算符号优先级 == 栈顶符合的优先级
                  直接从运算数栈中取出两个数，再取出栈顶符号。
                  把计算的结果入运算数栈，然后再把当前的运算符
                  入符号栈。
                  （例如：1 + 2 + 4 - 2中，1和2入数据栈，+入符号
                          栈，比较+和-优先级一样，则计算1+2= 3.
                          然后结果3入栈，4入栈，+入栈。再次计算）
                  3、当前的运算符号优先级 >栈顶符合的优先级
                          不要计算，把当前符号直接再次入栈，此时栈顶元素为
                          当前符号，然后再把数据入栈，当新的符号来的时候，
                          比较符号和栈顶元素，若是小于栈顶元素的话，则计算。
                  （例如：1 + 2 * 3-4,1和2入栈，+入栈，然后判断*和+的
                          优先级，* > + ,*入栈，3 入栈， 然后判断-和*
                          的优先级，则- < *,则计算2 和3出栈，*出栈，计算
                          2 * 3 = 6，然后6入栈，在计算+ 和 -的优先级，一样
                          则，计算1 + 6 = 7，然后7入栈，4入栈，-入栈，依次计算）

#include "link_stack.h"

// 1. 创建空的链式栈——为栈头在堆区分配空间
link_stack_t *create_empty_link_stack() {
    link_stack_t *s = NULL;
    s = (link_stack_t *) malloc(sizeof(link_stack_t));
    if (NULL == s) {
        printf("malloc failed!\n");
        return NULL;
    }
    memset(s, 0, sizeof(link_stack_t));
    return s;
}

// 2. 判断栈是否为空
int is_empty_link_stack(link_stack_t *s) {
    return s->top == NULL;
}

// 3. 入栈
void push_link_stack(link_stack_t *s, data_t data) {
    link_list_t *temp = NULL;
    temp = (link_list_t *) malloc(sizeof(link_stack_t));
    if (NULL == temp) {
        printf("malloc failed!\n");
        return;
    }
    temp->data = data;

    // 插入数据，类似于链表的头插法
    temp->next = s->top;
    s->top = temp;

    // 更新n的值
    s->n++;
}

// 4. 出栈
data_t pop_link_stack(link_stack_t *s) {
    link_list_t *temp = NULL;
    data_t data;

    // 1. 保存要删除结点的地址
    temp = s->top;

    // 2. 取出数据
    data = temp->data;

    // 3. 更新指针信息
    s->top = temp->next;

    // 4. 释放temp结点
    free(temp);
    temp = NULL;

    // 5. 更新n的值
    s->n--;

    return data;
}

// 5. 输出栈顶元素的值
data_t get_top_data(link_stack_t *s) {
    return s->top->data;
}

#ifndef EMBEDDED_LINK_STACK_H
#define EMBEDDED_LINK_STACK_H

#include <stdlib.h>
#include <string.h>
#include <stdio.h>

// 结点类型的设计
typedef int data_t;

typedef struct node {
    data_t data;
    struct node *next;
} link_list_t;

// 栈头类型的设计
typedef struct {
    link_list_t *top; // 栈顶指针
    int n;// 记录当前栈中元素的个数
} link_stack_t;

extern link_stack_t *create_empty_link_stack();

extern int is_empty_link_stack(link_stack_t *s);

extern void push_link_stack(link_stack_t *s, data_t data);

extern data_t pop_link_stack(link_stack_t *s);

extern data_t get_top_data(link_stack_t *s);

#endif //EMBEDDED_LINK_STACK_H

#include "link_stack.h"

// 获得优先级函数
int get_level(char operator) {
    switch (operator) {
        case '+':
        case '-':
            return 1;
        case '*':
        case '/':
            return 2;
        default:
            printf("Invalid operator : %c.", operator);
            return -1;
    }
}

// 计算+、-、*、/函数
// 参数说明：
// operand_stack 运算数
// operator_stack 运算符
int compute(link_stack_t *operand_stack, link_stack_t *operator_stack) {
    int data, data1, data2;
    char c = 0;
    data2 = pop_link_stack(operand_stack); // 运算数2出栈
    data1 = pop_link_stack(operand_stack); // 运算数1出栈

    c = (char) pop_link_stack(operator_stack); // 运算符出栈

    switch (c) {
        case '+':
            data = data1 + data2;
            break;
        case '-':
            data = data1 - data2;
            break;
        case '*':
            data = data1 * data2;
            break;
        case '/':
            data = data1 / data2;
            break;
        default:
            break;
    }


    push_link_stack(operand_stack, data); // 把最后获得的运算数入栈
}

// 处理运算符
// 参数说明：
// operand_stack 运算数
// operator_stack 运算符
void deal_with(link_stack_t *operand_stack, link_stack_t *operator_stack, const char *p) {
    // 如果它的优先级比运算符栈栈顶元素的优先级高或者空栈就进运算符栈
    if (is_empty_link_stack(operator_stack) || (
            *p != '\0' &&
            get_level(*p) > get_level((char) get_top_data(operator_stack))
    )
            ) {
        push_link_stack(operator_stack, *p);
    } else {
        // 计算
        compute(operand_stack, operator_stack);
        // 继续比较该运算符与栈顶运算符的优先级
        deal_with(operand_stack, operator_stack, p);
    }
}

int main() {
    char buf[1024];
    char *p = buf;
    int data = 0;
    int count = 0;
    int i = 0;
    int j = 0;
    int m = 1;

    printf("请输入运算表达式：\n");
    scanf("%s", buf);

    link_stack_t *operand_stack = create_empty_link_stack();// 运算数栈
    link_stack_t *operator_stack = create_empty_link_stack(); // 运算符栈

    while (1) {
        // 操作数一律进操作数栈
        if (*p >= '0' && *p <= '9') {
            // 若是连续的运算数,合并成一个完整的数data
            data = 0;
            count = 0;
            char *q = p;
            while (*q >= '0' && *q <= '9') {
                count++;
                q++;
            }
            for (i = count - 1; i >= 0; i--) {
                m = 1;
                for (j = 0; j < count - 1 - i; j++) {
                    m *= 10;
                }
                data += m * (int)(p[i] - 48);
            }
            if (count) {
                p = q - 1;
            }
            push_link_stack(operand_stack, data);
        }
        // 遇到运算符时
        if (*p == '+' || *p == '-' || *p == '*' || *p == '/') {
            // 处理运算符
            deal_with(operand_stack, operator_stack, p);
        }
        // 遇到\0时
        if (*p == '\0') {
            // 处理运算符
            deal_with(operand_stack, operator_stack, p);
            break;
        }
        p++;
    }

    printf("运算结果为：%d", pop_link_stack(operand_stack));
    return 0;
}
