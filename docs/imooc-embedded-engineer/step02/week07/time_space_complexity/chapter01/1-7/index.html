bi_tree.c

//
// Created by gaosx on 22-9-2.
//
#include "bi_tree.h"

bi_tree_t *create_binary_tree(int n) {
    bi_tree_t *root = NULL;

    root = (bi_tree_t *) malloc(sizeof(bi_tree_t));
    if (NULL == root) {
        printf("malloc failed!\n");
        return NULL;
    }
    memset(root, 0, sizeof(bi_tree_t));

    root->n = n;

    printf("Input %d node data: ", n);
    scanf("%c", &(root->data));

    // 清除缓冲区数据
    while (getchar() != '\n');

    // 左孩子存在条件
    if (2 * n <= N) {
        root->l_child = create_binary_tree(2 * n);
    }

    // 右孩子存在条件
    if (2 * n + 1 <= N) {
        root->r_child = create_binary_tree(2 * n + 1);
    }

    return root;
}

// 求二叉树的深度
int get_binary_tree_depth(bi_tree_t *root) {
    if (root == NULL) {
        return 0;
    }
    int left = get_binary_tree_depth(root->l_child);
    int right = get_binary_tree_depth(root->r_child);

    return left >= right ? left + 1 : right + 1;
}

bi_tree.h

//
// Created by gaosx on 22-9-2.
//

#ifndef EMBEDDED_BI_TREE_H
#define EMBEDDED_BI_TREE_H

#include <string.h>
#include <stdio.h>
#include <stdlib.h>

#define N 6

typedef char data_t;

typedef struct bi_tree {
    int n; // 保存编号
    data_t data; // 保存字符
    struct bi_tree *l_child; // 保存左孩子的地址
    struct bi_tree *r_child; // 保存右孩子的地址
} bi_tree_t;

extern bi_tree_t *create_binary_tree(int n);

extern int get_binary_tree_depth(bi_tree_t *root);

#endif //EMBEDDED_BI_TREE_H

main.c

//

int main() {

    root = create_binary_tree(1);

    printf("create is successfully!\n");
    printf("二叉树的深度为:%d", get_binary_tree_depth(root));
    return 0;
}

    linklist_t *head = NULL;

    // 1.1 分配堆区空间
    head = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == head) {
        printf("malloc failed!\n");
        return NULL;
    }

    memset(head, 0, sizeof(linklist_t));
    // head->next = NULL;
    // head->data = 0;
    return head;

// 尾插法
// 特点：插入的顺序和输出的顺序是相同的
void insert_tail_linklist(linklist_t *head, datatype_t data) {
    // 3.1 为结点在堆区申请空间
    linklist_t *temp = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == temp) {
        printf("malloc failed!\n");
    }

    // 3.2 插入数据
    temp->data = data;

    // 3.3 找到尾结点
    linklist_t *tail = head;
    while (tail->next != NULL) {
        tail = tail->next;
    }

    // 3.4 连接结点
    temp->next = tail->next;
    tail->next = temp;
}

// 输出链表中的内容
void print_data_linklist(linklist_t *head) {
    linklist_t *p = head;

    while (p->next != NULL) {
        printf("%d ", p->next->data);
        p = p->next;
    }
    printf("\n");
}

// 在单向链表中实现冒泡排序
void sort_linklist(linklist_t *head) {
    linklist_t *p = NULL;
    linklist_t *q = NULL;
    linklist_t *tail = NULL;

    while (head->next != tail) {
        p = head;
        while (p->next != tail && p->next->next != tail) {
            if (p->next->data > p->next->next->data) {
                q = p->next;
                p->next = p->next->next;
                q->next = p->next->next;
                p->next->next = q;
                q = NULL;
            }
            p = p->next;
        }
        tail = p->next;
    }
}

//
// Created by gaosx on 22-8-31.
//
#include "linklist.h"

// 1. 创建空的链表---为头结点在堆区分配空间
linklist_t *create_empty_linklist() {
    linklist_t *head = NULL;

    // 1.1 分配堆区空间
    head = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == head) {
        printf("malloc failed!\n");
        return NULL;
    }

    memset(head, 0, sizeof(linklist_t));
    // head->next = NULL;
    // head->data = 0;
    return head;
}

// 有序插入
// 特点：
void insert_order_linklist(linklist_t *head, datatype_t data) {
    // 4.1 为结点在堆区申请空间
    linklist_t *temp = (linklist_t *) malloc(sizeof(linklist_t));
    if (NULL == temp) {
        printf("malloc failed!\n");
    }

    // 3.2 插入数据
    temp->data = data;

    linklist_t *p = head;

    // 3.3 p指针向后遍历
    while (p->next != NULL && data > p->next->data) {
        p = p->next;
    }

    // 3.4 在p节点后插入temp节点
    temp->next = p->next;
    p->next = temp;
}

// 输出链表中的内容
void print_data_linklist(linklist_t *head) {
    linklist_t *p = head;

    while (p->next != NULL) {
        printf("%d ", p->next->data);
        p = p->next;
    }
    printf("\n");
}

// 将两个有序链表合成一个有序链表
linklist_t *and_linklist(linklist_t *head1, linklist_t *head2) {
    linklist_t *p = NULL;

    p = head2;
    while (p->next != NULL) {
        insert_order_linklist(head1, p->next->data);
        p = p->next;
    }
    return head1;
}

// 初始化链表
void init_linklist(linklist_t *head, char *name) {
    int data = 0;
    while (1) {
        printf("请输入%s链表中需要包含的元素:", name);
        scanf("%d", &data);
        printf("received:%d\n", data);
        insert_order_linklist(head, data);
        //清除缓冲区数据
        while(getchar() != '\n');
        printf("是否继续(y/n):");
        if (getchar() == 'n') {
            break;
        }
    }
}
