1-6 表达式求值升级版

在上一个题目的基础上，加上括号的操作。大家思考一下。

栈在表达式计算过程中的应用 ：建立操作数栈和运算符栈。运算符有优先级。规则：

自左至右扫描表达式，凡是遇到操作数一律进操作数栈。

当遇到运算符时，如果它的优先级比运算符栈栈顶元素的优先级高就进栈。反之，取出栈顶运算符和操作数栈栈顶的连续两个操作数进行运算，并将结果存入操作数栈，然后继续比较该运算符与栈顶运算符的优先级。

左括号一律进运算符栈，右括号一律不进运算符栈，取出运算符栈顶运算符和操作数栈顶的两个操作数进行运算，并将结果压入操作数栈，直到取出左括号为止。

#include "link_stack.h"

// 1. 创建空的链式栈——为栈头在堆区分配空间
link_stack_t *create_empty_link_stack() {
    link_stack_t *s = NULL;
    s = (link_stack_t *) malloc(sizeof(link_stack_t));
    if (NULL == s) {
        printf("malloc failed!\n");
        return NULL;
    }
    memset(s, 0, sizeof(link_stack_t));
    return s;
}

// 2. 判断栈是否为空
int is_empty_link_stack(link_stack_t *s) {
    return s->top == NULL;
}

// 3. 入栈
void push_link_stack(link_stack_t *s, data_t data) {
    link_list_t *temp = NULL;
    temp = (link_list_t *) malloc(sizeof(link_stack_t));
    if (NULL == temp) {
        printf("malloc failed!\n");
        return;
    }
    temp->data = data;

    // 插入数据，类似于链表的头插法
    temp->next = s->top;
    s->top = temp;

    // 更新n的值
    s->n++;
}

// 4. 出栈
data_t pop_link_stack(link_stack_t *s) {
    link_list_t *temp = NULL;
    data_t data;

    // 1. 保存要删除结点的地址
    temp = s->top;

    // 2. 取出数据
    data = temp->data;

    // 3. 更新指针信息
    s->top = temp->next;

    // 4. 释放temp结点
    free(temp);
    temp = NULL;

    // 5. 更新n的值
    s->n--;

    return data;
}

// 5. 输出栈顶元素的值
data_t get_top_data(link_stack_t *s) {
    return s->top->data;
}

#ifndef EMBEDDED_LINK_STACK_H
#define EMBEDDED_LINK_STACK_H

#include <stdlib.h>
#include <string.h>
#include <stdio.h>

// 结点类型的设计
typedef int data_t;

typedef struct node {
    data_t data;
    struct node *next;
} link_list_t;

// 栈头类型的设计
typedef struct {
    link_list_t *top; // 栈顶指针
    int n;// 记录当前栈中元素的个数
} link_stack_t;

extern link_stack_t *create_empty_link_stack();

extern int is_empty_link_stack(link_stack_t *s);

extern void push_link_stack(link_stack_t *s, data_t data);

extern data_t pop_link_stack(link_stack_t *s);

extern data_t get_top_data(link_stack_t *s);

#endif //EMBEDDED_LINK_STACK_H

//
// Created by gaosx on 22-9-10.
//

#include "link_stack.h"

// 获得优先级函数
int get_level(char operator) {
    switch (operator) {
        case '+':
        case '-':
            return 1;
        case '*':
        case '/':
            return 2;
        case '(':
            return 0;
        default:
            printf("Invalid operator : %c.", operator);
            return -1;
    }
}

// 计算+、-、*、/函数
// 参数说明：
// operand_stack 运算数
// operator_stack 运算符
int compute(link_stack_t *operand_stack, link_stack_t *operator_stack) {
    int data, data1, data2;
    char c = 0;
    c = (char) pop_link_stack(operator_stack); // 运算符出栈
    if (c == '(') {
        return -1;
    }

    data2 = pop_link_stack(operand_stack); // 运算数2出栈
    data1 = pop_link_stack(operand_stack); // 运算数1出栈


    switch (c) {
        case '+':
            data = data1 + data2;
            break;
        case '-':
            data = data1 - data2;
            break;
        case '*':
            data = data1 * data2;
            break;
        case '/':
            data = data1 / data2;
            break;
        default:
            break;
    }


    push_link_stack(operand_stack, data); // 把最后获得的运算数入栈
    return 0;
}

// 处理运算符
// 参数说明：
// operand_stack 运算数
// operator_stack 运算符
void deal_with(link_stack_t *operand_stack, link_stack_t *operator_stack, const char *p) {
    // 如果它的优先级比运算符栈栈顶元素的优先级高或者空栈就进运算符栈
    // 左括号一律进运算符栈
    if (is_empty_link_stack(operator_stack) || *p == '(' || (
            *p != '\0' &&
            *p != ')' &&
            get_level(*p) > get_level((char) get_top_data(operator_stack))
    )
            ) {
        push_link_stack(operator_stack, *p);
    } else {
        // 计算
        int r = compute(operand_stack, operator_stack);
        if (r == -1) {
            return;
        }
        // 继续比较该运算符与栈顶运算符的优先级
        deal_with(operand_stack, operator_stack, p);
    }
}

int main() {
    char buf[1024];
    char *p = buf;
    int data = 0;
    int count = 0;
    int i = 0;
    int j = 0;
    int m = 1;

    printf("请输入运算表达式：\n");
    scanf("%s", buf);

    link_stack_t *operand_stack = create_empty_link_stack();// 运算数栈
    link_stack_t *operator_stack = create_empty_link_stack(); // 运算符栈

    while (1) {
        // 操作数一律进操作数栈
        if (*p >= '0' && *p <= '9') {
            // 若是连续的运算数,合并成一个完整的数data
            data = 0;
            count = 0;
            char *q = p;
            while (*q >= '0' && *q <= '9') {
                count++;
                q++;
            }
            for (i = count - 1; i >= 0; i--) {
                m = 1;
                for (j = 0; j < count - 1 - i; j++) {
                    m *= 10;
                }
                data += m * (int)(p[i] - 48);
            }
            if (count) {
                p = q - 1;
            }
            push_link_stack(operand_stack, data);
        }
        // 遇到运算符时
        if (*p == '+' || *p == '-' || *p == '*' || *p == '/' || *p == '(' || *p == ')') {
            // 处理运算符
            deal_with(operand_stack, operator_stack, p);
        }
        // 遇到\0时
        if (*p == '\0') {
            // 处理运算符
            deal_with(operand_stack, operator_stack, p);
            break;
        }
        p++;
    }

    printf("运算结果为：%d", pop_link_stack(operand_stack));
    return 0;
}
